# 976三角形周长

/**
 * @param {number[]} A
 * @return {number}
 */

// 先排序从大到小
const getSort = (A) => {
	return A.sort((a,b)=> (b-a));
}
const largestPerimeter = function(A) {
	const _sortArr = getSort(A);
    // 从降序最大边开始
    for(let i =0  ; i< _sortArr.length - 2 ; i++) {
        if(_sortArr[i+2] + _sortArr[i+1] > _sortArr[i]) {
            return _sortArr[i] +_sortArr[i+1]+ _sortArr[i+2];
        }
    }
    return 0;
};
